-- Lógica para atualizar a tela de alvos ativos
local function UpdateCustomListDisplay()
    local names = {}
    for fish, active in pairs(PeixesAlvo) do
        if active then
            table.insert(names, fish)
        end
    end
    if #names == 0 then
        CustomListDisplay.Text = "Current targets: (None)"
    else
        table.sort(names)
        CustomListDisplay.Text = "Current targets: " .. table.concat(names, ", ")
    end
end

-- Ativadores Rápidos de Raridade
local function ToggleRarityGroup(rarityName)
    local anyActiveOfThisRarity = false
    for _, item in ipairs(FISH_DATABASE) do
        if item.Rarity == rarityName and PeixesAlvo[item.Name] then
            anyActiveOfThisRarity = true
            break
        end
    end

    for _, item in ipairs(FISH_DATABASE) do
        if item.Rarity == rarityName then
            PeixesAlvo[item.Name] = not anyActiveOfThisRarity or nil
        end
    end
    UpdateCustomListDisplay()
end

ToggleMythic.MouseButton1Click:Connect(function() ToggleRarityGroup("Mythic") end)

ToggleRare.MouseButton1Click:Connect(function() ToggleRarityGroup("Rare") end)
ToggleCommon.MouseButton1Click:Connect(function() ToggleRarityGroup("Common") end)

-- Limpar Dropdown
local function ClearDropdown()
    for _, child in ipairs(DropdownFrame:GetChildren()) do
        if child:IsA("TextButton") then child:Destroy() end
    end
    DropdownFrame.Visible = false
end

-- Sistema Inteligente: Filtro ao Digitar + Auto-Complete
CustomFishTextBox:GetPropertyChangedSignal("Text"):Connect(function()
    local input = string.lower(CustomFishTextBox.Text)
    ClearDropdown()
    
    if input == "" then return end
    
    local matches = {}
    for _, fishData in ipairs(FISH_DATABASE) do
        if string.find(string.lower(fishData.Name), input) then
            table.insert(matches, fishData)
        end
    end
    
    if #matches > 0 then
        DropdownFrame.Visible = true

local function ParseTimeToSeconds(text)
    if not text then return 0 end
    text = string.lower(text)
    local minutes = string.match(text, "(%d+)%s*m")
    local seconds = string.match(text, "(%d+)%s*s")
    local totalSeconds = 0
    if minutes then totalSeconds = totalSeconds + (tonumber(minutes) * 60) end
    if seconds then totalSeconds = totalSeconds + tonumber(seconds)
    elseif not minutes then
        local cleanNum = string.match(text, "(%d+)")
        if cleanNum then totalSeconds = tonumber(cleanNum) end
    end
    return totalSeconds
end

local function FormatSecondsToText(seconds)
    if seconds <= 0 then return "Unlocked" end
    if seconds < 60 then return tostring(seconds) .. "s"
    else
        local mins = math.floor(seconds / 60)
        local secs = seconds % 60
        return string.format("%dm %02ds", mins, secs)
    end
end

local function ExecuteLockTeleport(forceBuyModel)
    if not forceBuyModel or isTeleporting then return end
    isTeleporting = true
    task.spawn(function()
        local lock1 = forceBuyModel:FindFirstChild("Lock", true)
        local lockPart = lock1 and lock1:FindFirstChild("Lock", true) or lock1
        local char = LocalPlayer.Character
        local hum = char and char:FindFirstChildOfClass("Humanoid")
        
        if lockPart and lockPart:IsA("BasePart") and char and char:FindFirstChild("HumanoidRootPart") and hum then
            local hrp = char.HumanoidRootPart
            local previousCFrame = hrp.CFrame 
            local oldWalkSpeed = hum.WalkSpeed
            local oldJumpPower = hum.JumpPower
            
            hum.WalkSpeed = 0
            hum.JumpPower = 0
            
            hrp.AssemblyLinearVelocity = Vector3.zero
            hrp.AssemblyAngularVelocity = Vector3.zero
            hrp.CFrame = lockPart.CFrame * CFrame.new(0, 3, 0) 
            
            task.wait(2.0)
            if char and char:FindFirstChild("HumanoidRootPart") then
                char.HumanoidRootPart.CFrame = previousCFrame 
            end
            if hum then
                hum.WalkSpeed = walkspeedEnabled and currentSpeed or 16
                hum.JumpPower = oldJumpPower
            end
        end
        task.wait(1.0)
        isTeleporting = false
    end)
end

        if LocalPlayer.Character then setupAntiKB(LocalPlayer.Character) end
    else
        AntiRagdollButton.Text = "Anti-Ragdoll & KB: OFF"
        AntiRagdollButton.BackgroundColor3 = FromRGB(88, 101, 242)
        if ragdollConnection then ragdollConnection:Disconnect() end
        for _, conn in ipairs(kbConnections) do conn:Disconnect() end
        table.clear(kbConnections)
    end
end)

-- Inf Jump
local infJumpEnabled = false
local infJumpConnection = nil
InfJumpButton.MouseButton1Click:Connect(function()
    infJumpEnabled = not infJumpEnabled
    if infJumpEnabled then
        InfJumpButton.Text = "Infinite Jump: ON"
        InfJumpButton.BackgroundColor3 = FromRGB(67, 181, 129)
        infJumpConnection = UserInputService.JumpRequest:Connect(function()
            local char = LocalPlayer.Character
            if char then
                local hum = char:FindFirstChildOfClass("Humanoid")
                if hum then hum:ChangeState(Enum.HumanoidStateType.Jumping) end
            end
        end)
    else
        InfJumpButton.Text = "Infinite Jump: OFF"
        InfJumpButton.BackgroundColor3 = FromRGB(88, 101, 242)
        if infJumpConnection then infJumpConnection:Disconnect() end
    end
end)

-- Walkspeed
local wsConnection = nil
WalkspeedToggle.MouseButton1Click:Connect(function()
    walkspeedEnabled = not walkspeedEnabled
    if walkspeedEnabled then
        WalkspeedToggle.Text = "Walkspeed: ON"
        WalkspeedToggle.BackgroundColor3 = FromRGB(67, 181, 129)
        wsConnection = RunService.Heartbeat:Connect(function()
            local char = LocalPlayer.Character
            if char then
                local hum = char:FindFirstChildOfClass("Humanoid")
                if hum and hum.WalkSpeed ~= 0 and not isTeleporting then hum.WalkSpeed = currentSpeed end
            end
        end)
    else
        WalkspeedToggle.Text = "Walkspeed: OFF"
        WalkspeedToggle.BackgroundColor3 = FromRGB(88, 101, 242)
        if wsConnection then wsConnection:Disconnect() end
        local char = LocalPlayer.Character
        if char then
            local hum = char:FindFirstChildOfClass("Humanoid")
            if hum then hum.WalkSpeed = 16 end
        end
    end
end)

local dragging = false
local function UpdateSlider()
    local relX = Mouse.X - WalkspeedSliderBG.AbsolutePosition.X
    local percent = math.clamp(relX / WalkspeedSliderBG.AbsoluteSize.X, 0, 1)
    WalkspeedSliderFill.Size = UDim2.new(percent, 0, 1, 0)
    currentSpeed = math.floor(16 + (percent * 184))
    WalkspeedValueLabel.Text = "Speed: " .. currentSpeed
    if walkspeedEnabled then
        local char = LocalPlayer.Character
        local hum = char and char:FindFirstChildOfClass("Humanoid")
        if hum and not isTeleporting then hum.WalkSpeed = currentSpeed end
    end
end

WalkspeedSliderBG.MouseButton1Down:Connect(function() dragging = true; UpdateSlider() end)
Mouse.Move:Connect(function() if dragging then UpdateSlider() end end)
UserInputService.InputEnded:Connect(function(input) if input.UserInputType == Enum.UserInputType.MouseButton1 then dragging = false end end)
