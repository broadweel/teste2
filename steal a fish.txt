local function UpdateCustomListDisplay()
    local names = {}
    for fish, active in pairs(PeixesAlvo) do if active then table.insert(names, fish) end end
    if #names == 0 then CustomListDisplay.Text = "Current targets: (None)"
    else table.sort(names); CustomListDisplay.Text = "Current targets: " .. table.concat(names, ", ") end
end

local function ToggleRarityGroup(rarityName)
    local anyActive = false
    for _, item in ipairs(FISH_DATABASE) do if item.Rarity == rarityName and PeixesAlvo[item.Name] then anyActive = true; break end end
    for _, item in ipairs(FISH_DATABASE) do if item.Rarity == rarityName then PeixesAlvo[item.Name] = not anyActive or nil end end
    UpdateCustomListDisplay()
end

local function ClearDropdown()
    for _, child in ipairs(DropdownFrame:GetChildren()) do if child:IsA("TextButton") then child:Destroy() end end
    DropdownFrame.Visible = false
end

CustomFishTextBox:GetPropertyChangedSignal("Text"):Connect(function()
    local input = string.lower(CustomFishTextBox.Text)
    ClearDropdown()
    if input == "" then return end
    
    local matches = {}
    for _, f in ipairs(FISH_DATABASE) do if string.find(string.lower(f.Name), input) then table.insert(matches, f) end end
    
    if #matches > 0 then
        DropdownFrame.Visible = true
        local limit = math.min(#matches, 5)

-- AUTO TP & LOCK LOGIC
local function ParseTimeToSeconds(text)
    if not text then return 0 end
    text = string.lower(text)
    local minutes = string.match(text, "(%d+)%s*m")
    -- Modificado para ler decimais (ex: 0.1s)
    local seconds = string.match(text, "(%d+%.?%d*)%s*s") 
    local total = 0
    if minutes then total = total + (tonumber(minutes) * 60) end
    if seconds then total = total + tonumber(seconds)
    elseif not minutes then
        local cleanNum = string.match(text, "(%d+%.?%d*)")
        if cleanNum then total = tonumber(cleanNum) end
    end
    return total
end

local function FormatSecondsToText(seconds)
    if seconds <= 0 then return "Unlocked" end
    if seconds < 60 then return tostring(seconds) .. "s" end
    return string.format("%dm %02ds", math.floor(seconds / 60), seconds % 60)
end

local function ExecuteLockTeleport(forceBuyModel)
    if not forceBuyModel or isTeleporting then return end
    isTeleporting = true
    task.spawn(function()
        
        -- FIX 3: Se estiver invisível, ele desliga automaticamente antes de puxar para comprar
        if invisEnabled then
            SetInvisState(false)
            task.wait(0.2)
        end

        local lock1 = forceBuyModel:FindFirstChild("Lock", true)
        local lockPart = lock1 and lock1:FindFirstChild("Lock", true) or lock1
        local char = LocalPlayer.Character
        local hum = char and char:FindFirstChildOfClass("Humanoid")
        
        if lockPart and lockPart:IsA("BasePart") and char and char:FindFirstChild("HumanoidRootPart") and hum then
            local hrp = char.HumanoidRootPart
            local previousCFrame = hrp.CFrame 
            local oldWalkSpeed = hum.WalkSpeed
            local oldJumpPower = hum.JumpPower
            
            hum.WalkSpeed = 0
            hum.JumpPower = 0
            hrp.AssemblyLinearVelocity = Vector3.zero
            hrp.AssemblyAngularVelocity = Vector3.zero
            hrp.CFrame = lockPart.CFrame * CFrame.new(0, 3, 0) 
            
            task.wait(3.0)
            if char and char:FindFirstChild("HumanoidRootPart") then char.HumanoidRootPart.CFrame = previousCFrame end
            if hum then
                hum.WalkSpeed = walkspeedEnabled and currentSpeed or 16
                hum.JumpPower = oldJumpPower
            end
        end
        task.wait(1.0)
        isTeleporting = false
    end)
end
